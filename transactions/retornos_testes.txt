4 - Transaction

4.1 addTransactionToList
	0 = Transação adicionada com sucesso
	1 = Ponteiro nulo recebido
-1 = Erro de alocação de memória

4.2 makeTransactionSaving
	0 = Transação realizada com sucesso
	1 = Valor inválido
	2 = CPF inválido
 	3 = CPF não encontrado
	4 = Tipo de transação inválida
	5 =  Saldo insuficiente
-1 = Erro de alocação de memória

4.3 getTransactions
	0 = Transacções encontradas
	1 = CPF inválido
	2 = CPF não encontrado
	3 = CPF sem transações
	
4.4 addTransactionListToList
          0 = Transação adicionada com sucesso
          1 = Ponteiro Inválido
          2 = Nó mal formulado

4.5 makeTransactionChecking
	0 = Transação realizada com sucesso
	1 = Valor inválido
	2 = CPF target inválido
	3 = CPF source inválido
 	4 = CPF source não encontrado
	5 = CPF target não encontrado
-1 = Erro de alocação de memória


4.6 makeTransactionSaving
	0 = Transação realizada com sucesso

 	3 = CPF não encontrado

4.8 saveTransactions
